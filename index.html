¿Cómo obtengo el archivo con todos los registros?
  1. Abre la página oficial: resultados-simulacro.html
  2. Abre las herramientas de desarrollador (F12) y ve a la pestaña Consola.
  3. Pega el script de abajo y presiona Enter. Recorrerá las 82 páginas del listado (tarda ~1 minuto) y descargará automáticamente resultados_simulacro_uni.json.
  4. Vuelve aquí y carga ese archivo con el botón "Cargar archivo".
(async function scrapeUNI(){
  const sleep = ms => new Promise(r => setTimeout(r, ms));
  function getTotalPages(){
    const m = document.body.innerText.match(/P[aá]gina\s+\d+\s+de\s+(\d+)/i);
    return m ? parseInt(m[1], 10) : 82;
  }
  function getRows(){
    return Array.from(document.querySelectorAll('table tbody tr')).map(tr =>
      Array.from(tr.querySelectorAll('td')).map(td => td.textContent.trim())
    ).filter(c => c.length >= 3);
  }
  function findNextButton(){
    return Array.from(document.querySelectorAll('button, a, [role="button"]'))
      .find(el => el.textContent.trim().toUpperCase().includes('SIGUIENTE') && !el.disabled);
  }
  const seen = new Map();
  const totalPages = getTotalPages();
  let page = 1, stagnant = 0;
  while (page <= totalPages && stagnant < 6) {
    await sleep(300);
    const before = seen.size;
    getRows().forEach(r => seen.set(r[0], r));
    stagnant = (seen.size === before) ? stagnant + 1 : 0;
    const btn = findNextButton();
    if (!btn) break;
    btn.click();
    page++;
    await sleep(500);
  }
  const data = Array.from(seen.values()).map(([codigo, nombres, puntajeRaw]) => ({
    codigo, nombres, puntaje: parseFloat(String(puntajeRaw).replace(',', '.'))
  }));
  console.log('Total capturado:', data.length);
  const blob = new Blob([JSON.stringify(data, null, 2)], {type:'application/json'});
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url; a.download = 'resultados_simulacro_uni.json';
  document.body.appendChild(a); a.click(); a.remove();
  console.log('Descarga iniciada: resultados_simulacro_uni.json');
})();

Si la página cambia su estructura y el script no encuentra las filas o el botón "SIGUIENTE", el conteo se detiene solo; puedes ajustar los selectores o decírmelo y lo adapto.
